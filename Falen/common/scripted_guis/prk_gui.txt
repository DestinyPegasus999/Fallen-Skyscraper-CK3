prk_gui_toggle = {
	scope = character
	
	is_shown = {
		has_variable = prk_gui_toggle
	}

	effect = {
		if = {
			limit = {
				has_variable = prk_gui_toggle
			}
			remove_variable = prk_gui_toggle
		}
		else = {
			set_variable = prk_gui_toggle
		}
	}
}

prk_gui_close = {
	scope = character
	
	is_shown = {
		always = yes
	}

	effect = {
		remove_variable = prk_gui_toggle
	}
}
prk_gui_abc = {
	scope = character
	is_shown = {
		has_landed_title = k_prk
	}
}
prk_get_president = {
	scope = character
	effect = {
		title:k_prk = {
			holder = {
				root = {
					set_variable = {
						name = prk_president
						 value = prev
					}
				}
			}
		}
	}
}
prk_candidate = {
	scope = character
	effect = {
		every_living_character = {
			limit = { has_variable = candidate }
			root = {
				add_variable_to_list = {
					name = prk_candidates
					target = prev
				}
			}
		}
	}
}
prk_run_candidate = {
	scope = character
	is_valid = {
		root = { has_variable = candidate }
	}
	effect = {
		root = {
			set_variable = candidate
		}
	}
}
prk_run_candidate = {
	scope = character
	is_valid = {
		root = { NOT = { has_variable = candidate } }
	}
	effect = {
		root = {
			remove_variable = candidate
		}
	}
}
